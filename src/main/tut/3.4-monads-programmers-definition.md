implicit def writerMonad[W: Monoid] =
  new Monad[({type T[A] = Writer[W, A]})#T] {

    def >=>[A, B, C](
        f: A => Writer[W, B],
        g: B => Writer[W, C]) =
      a => {
        val Writer((b, s1)) = f(a)
        val Writer((c, s2)) = g(b)
        Writer((c, Monoid[W].combine(s1, s2)))
      }

    def pure[A](a: A) =
      Writer(a, Monoid[W].empty)
  }
