implicit def writerMonad[W: Monoid] =
  new Monad[({type T[A] = Writer[W, A]})#T] {

    def flatMap[A, B]
        (wa: Writer[W, A])
        (k: A => Writer[W, B])
        : Writer[W, B] = wa match {
      case Writer((a, w)) =>
        val ((a1, w1)) = runWriter(k(a))
